\DoxyCodeLine{\#\ MongoDB\ Configuration}
\DoxyCodeLine{MONGODB\_USER=my\_mongodb\_user}

\DoxyCodeLine{MONGODB\_DATABASE=my\_mongodb\_database}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ RabbitMQ\ Configuration}
\DoxyCodeLine{RABBITMQ\_USER=my\_rabbitmq\_user}
\DoxyCodeLine{RABBITMQ\_PASSWORD=my\_rabbitmq\_password}
\DoxyCodeLine{RABBITMQ\_MANAGEMENT\_PORT=15672}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ Nginx\ Configuration}
\DoxyCodeLine{NGINX\_PORT=8080}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ Spring\ Configuration}
\DoxyCodeLine{SPRING\_PORT=8081}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ React\ Configuration}
\DoxyCodeLine{FRONTEND\_PORT=8082}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ Email\ Configuration}
\DoxyCodeLine{EMAIL=my\_email@example.com}
\DoxyCodeLine{EMAIL\_PASSWORD=my\_email\_password}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ API\ Key\ Configuration}
\DoxyCodeLine{API\_KEY=your\_api\_key}
\DoxyCodeLine{}
\DoxyCodeLine{FETCH\_INTERVAL\ =\ 5\ \#\ in\ minutes}
\DoxyCodeLine{}
\DoxyCodeLine{FRONTEND\_PORT\ =\ 8383}
\DoxyCodeLine{}

\end{DoxyCode}
